# 关卡设计

```json
[
  {
    "stage": 1,
    "limit": {
      // 生命不限
      "lives": -1,
      // 步数不限
      "steps": -1,
      // 时间不限
      "seconds": -1
    },
    "goal": {
      "score": 1000,
      // 同时存在 2 1 个，3 2 个
      "exist": [[2, 1], [3, 2]],
      // 合成 2 1 次，3 2 次
      "merge": [[2, 1], [3, 2]],
      // 2 连击 1 次，3 连击 2 次
      "combo": [[2, 1], [3, 2]],
      // 合成魔法数字 2 次
      "magic": 2
    }
  }
]
```
